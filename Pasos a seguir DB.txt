import java.sql.*;

public class Main {
    public static void main(String[] args) {
        Connection conn = null;
        try {
            // Paso 1: Registrar el driver JDBC
            Class.forName("com.mysql.cj.jdbc.Driver");

            // Paso 2: Establecer la conexión con la base de datos
            String url = "jdbc:mysql://localhost:3306/nombre_de_la_base_de_datos";
            String usuario = "tu_usuario";
            String contraseña = "tu_contraseña";
            conn = DriverManager.getConnection(url, usuario, contraseña);

            // Paso 3: Crear la sentencia SQL
            String sql = "SELECT * FROM nombre_de_la_tabla";
            Statement statement = conn.createStatement();

            // Paso 4: Ejecutar la consulta
            ResultSet resultSet = statement.executeQuery(sql);

            // Paso 5: Procesar los resultados
            while (resultSet.next()) {
                // Aquí puedes acceder a los datos de cada fila
                int id = resultSet.getInt("id");
                String nombre = resultSet.getString("nombre");
                // Puedes continuar con otros campos de tu tabla
                System.out.println("ID: " + id + ", Nombre: " + nombre);
            }
        } catch (SQLException ex) {
            ex.printStackTrace();
        } catch (ClassNotFoundException ex) {
            ex.printStackTrace();
        } finally {
            // Paso 6: Cerrar la conexión
            try {
                if (conn != null) {
                    conn.close();
                }
            } catch (SQLException ex) {
                ex.printStackTrace();
            }
        }
    }
}
